<script>
function createCarousel(images) {
  // 创建容器
  var root = document.currentScript.parentElement;

  // 注入 HTML 结构（不使用 id，全部局部）
  root.innerHTML = `
    <div style="
      position: relative;
      width: 80%;
      aspect-ratio: 16 / 9;
      background: #ccc;
      margin-bottom: 10px;
      border-radius: 10px;
      max-width: 800px;">
      
      <img class="carousel-img"
           style="width:100%;height:100%;object-fit:contain;background:#ccc;border-radius:10px;">
      
      <button class="btn-prev"
              style="position:absolute;left:10px;top:50%;transform:translateY(-50%);
              background:rgba(0,0,0,0.5);color:white;border:none;width:35px;height:35px;
              border-radius:50%;cursor:pointer;font-size:20px;">‹</button>
      
      <button class="btn-next"
              style="position:absolute;right:10px;top:50%;transform:translateY(-50%);
              background:rgba(0,0,0,0.5);color:white;border:none;width:35px;height:35px;
              border-radius:50%;cursor:pointer;font-size:20px;">›</button>
      
      <div class="dots"
           style="position:absolute;bottom:10px;left:50%;transform:translateX(-50%);
           display:flex;gap:8px;"></div>
    </div>
  `;

  // 绑定局部节点
  var imgTag = root.querySelector(".carousel-img");
  var dotsContainer = root.querySelector(".dots");
  var current = 0;

  // 建圆点
  images.forEach((src, i) => {
    var dot = document.createElement("span");
    dot.style.cssText =
      "width:8px;height:8px;border-radius:50%;background:white;opacity:0.5;cursor:pointer;";
    dot.onclick = () => showImage(i);
    dotsContainer.appendChild(dot);
  });

  // 显示图片
  function showImage(n) {
    current = n;
    imgTag.src = images[n];
    [...dotsContainer.children].forEach((d, i) =>
      d.style.opacity = i === n ? "1" : "0.5"
    );
  }

  // 左右按钮
  root.querySelector(".btn-prev").onclick = () =>
    showImage((current - 1 + images.length) % images.length);

  root.querySelector(".btn-next").onclick = () =>
    showImage((current + 1) % images.length);

  // 初始显示
  showImage(0);
}
</script>
