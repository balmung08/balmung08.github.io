<script>
// 你只需要改这里：给每个控件定义图片列表
var imageSets = {

  iros: [
  '/pic/IROS2025/1.jpg',
  '/pic/IROS2025/2.jpg',
  '/pic/IROS2025/3.jpg',
  '/pic/IROS2025/4.jpg',
  '/pic/IROS2025/5.jpg',
  '/pic/IROS2025/6.jpg',
  '/pic/IROS2025/7.JPG',
  '/pic/IROS2025/8.jpg'
  ],

  bechelor: [
    '/pic/Bechelor Graduate/1.jpg',
    '/pic/Bechelor Graduate/2.jpg',
    '/pic/Bechelor Graduate/3.jpg',
    '/pic/Bechelor Graduate/4.jpg',
    '/pic/Bechelor Graduate/5.jpg',
    '/pic/Bechelor Graduate/6.jpg',
  ],

};


// =====================================================
// 自动为每个 .carousel 生成轮播控件（无需手写结构）
// =====================================================
document.querySelectorAll(".carousel").forEach(function(wrapper) {

  let setName = wrapper.getAttribute("data-set");
  let images = imageSets[setName];
  let current = 0;

  // 创建结构
  wrapper.innerHTML = `
    <div class="box">
      <img class="img">
      <button class="prev">‹</button>
      <button class="next">›</button>
      <div class="dots"></div>
    </div>
  `;

  let imgTag = wrapper.querySelector(".img");
  let dots = wrapper.querySelector(".dots");

  // 生成圆点
  images.forEach((_, i) => {
    let d = document.createElement("span");
    d.onclick = () => show(i);
    dots.appendChild(d);
  });

  function show(n) {
    current = n;
    imgTag.src = images[n];
    [...dots.children].forEach((dot, i) =>
      dot.style.opacity = i === n ? "1" : "0.5");
  }

  wrapper.querySelector(".prev").onclick = () =>
    show((current - 1 + images.length) % images.length);

  wrapper.querySelector(".next").onclick = () =>
    show((current + 1) % images.length);

  show(0);
});
</script>
